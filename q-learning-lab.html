Reinforcement Learning Mini‑Lab

A compact, self‑contained tutorial: Q‑learning, SARSA, and Double Q‑learning on a tiny GridWorld. White background, thin typography, and collapsible code blocks.

Overview

Goal: Learn by trial and error to maximize long‑term reward.

  • Agent interacts with an Environment in discrete steps.
  • At state s choose action a, receive reward r, move to s'.
  • Q‑learning: off‑policy value learning using the Bellman optimality update.
  • SARSA: on‑policy value learning using the action actually taken next.
  • Double Q‑learning: mitigates over‑estimation with two value tables.

#!/usr/bin/env python3
import os
import argparse
import numpy as np
import matplotlib.pyplot as plt

class GridWorld:
    def __init__(self, width=5, height=5, start=(0,0), goal=(4,4),
                 walls={(1,1),(1,2),(1,3)}, pits={(3,3)}, slip=0.1, step_cost=-1.0,
                 goal_reward=20.0, pit_reward=-10.0, seed=0):
        self.width = width
        self.height = height
        self.start = start
        self.goal = goal
        self.walls = set(walls)
        self.pits = set(pits)
        self.slip = slip
        self.step_cost = step_cost
        self.goal_reward = goal_reward
        self.pit_reward = pit_reward
        self.rng = np.random.default_rng(seed)
        self.n_actions = 4
        self.n_states = width * height
        self.pos = start

    def state_id(self, pos):
        x, y = pos
        return y * self.width + x

    def id_to_pos(self, sid):
        x = sid % self.width
        y = sid // self.width
        return (x, y)

    def reset(self):
        self.pos = self.start
        return self.state_id(self.pos)

    def apply_action(self, pos, a):
        x, y = pos
        if a == 0:   # up
            y -= 1
        elif a == 1: # right
            x += 1
        elif a == 2: # down
            y += 1
        elif a == 3: # left
            x -= 1
        return (x, y)

    def step(self, a):
        if self.rng.random() < self.slip:
            a = self.rng.integers(0, self.n_actions)
        nx, ny = self.apply_action(self.pos, a)
        nx = min(max(nx, 0), self.width-1)
        ny = min(max(ny, 0), self.height-1)
        if (nx, ny) in self.walls:
            nx, ny = self.pos
        self.pos = (nx, ny)
        reward = self.step_cost
        done = False
        if self.pos in self.pits:
            reward = self.pit_reward
            done = True
        elif self.pos == self.goal:
            reward = self.goal_reward
            done = True
        return self.state_id(self.pos), reward, done, {}

    def render_policy(self, Q=None):
        arrows = {0:'↑',1:'→',2:'↓',3:'←'}
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x,y) in self.walls:
                    row.append('■')
                elif (x,y) in self.pits:
                    row.append('X')
                elif (x,y) == self.goal:
                    row.append('G')
                elif (x,y) == self.start and Q is None:
                    row.append('S')
                else:
                    if Q is None:
                        row.append('·')
                    else:
                        a = int(np.argmax(Q[self.state_id((x,y))]))
                        row.append(arrows[a])
            grid.append(row)
        return grid

def q_learning(env, episodes=800, alpha=0.1, gamma=0.99, eps_start=1.0,
               eps_end=0.05, eps_decay=0.995, max_steps=100, seed=0):
    rng = np.random.default_rng(seed)
    Q = np.zeros((env.n_states, env.n_actions), dtype=np.float32)
    eps = eps_start
    rewards = []
    for ep in range(episodes):
        s = env.reset()
        total = 0.0
        for t in range(max_steps):
            if rng.random() < eps:
                a = rng.integers(0, env.n_actions)
            else:
                a = int(np.argmax(Q[s]))
            s2, r, done, _ = env.step(a)
            Q[s, a] = (1-alpha)*Q[s, a] + alpha*(r + gamma*np.max(Q[s2]))
            s = s2
            total += r
            if done:
                break
        rewards.append(total)
        eps = max(eps_end, eps * eps_decay)
    return Q, rewards

def moving_average(x, w):
    x = np.asarray(x, dtype=np.float64)
    if len(x) < w:
        return np.array([])
    return np.convolve(x, np.ones(w)/w, mode='valid')

def rollout(env, Q, max_steps=50):
    s = env.reset()
    path = [env.pos]
    total = 0.0
    for t in range(max_steps):
        a = int(np.argmax(Q[s]))
        s, r, done, _ = env.step(a)
        path.append(env.pos)
        total += r
        if done:
            break
    return path, total, done

def main():
    import argparse
    parser = argparse.ArgumentParser(description="Tiny Q-learning on a 5x5 GridWorld")
    parser.add_argument("--episodes", type=int, default=800)
    parser.add_argument("--alpha", type=float, default=0.1)
    parser.add_argument("--gamma", type=float, default=0.99)
    parser.add_argument("--eps_start", type=float, default=1.0)
    parser.add_argument("--eps_end", type=float, default=0.05)
    parser.add_argument("--eps_decay", type=float, default=0.995)
    parser.add_argument("--slip", type=float, default=0.1)
    parser.add_argument("--seed", type=int, default=0)
    args = parser.parse_args()

    env = GridWorld(slip=args.slip, seed=args.seed)
    Q, rewards = q_learning(env, episodes=args.episodes, alpha=args.alpha, gamma=args.gamma,
                            eps_start=args.eps_start, eps_end=args.eps_end, eps_decay=args.eps_decay)

    print("Learned greedy policy (arrows). Legend: ■=wall, X=pit, G=goal\\n")
    policy_grid = env.render_policy(Q)
    for row in policy_grid:
        print(" ".join(row))

    path, total, done = rollout(env, Q)
    print("\\nGreedy rollout from S to terminal:")
    print(" -> ".join([str(p) for p in path]))
    print(f"Total reward: {total:.1f} | Reached terminal: {done}")

    ma = moving_average(rewards, 50)
    plt.figure()
    plt.plot(rewards, label="Episode reward")
    if len(ma) > 0:
        start_idx = 50 - 1
        plt.plot(range(start_idx, start_idx + len(ma)), ma, label="Moving average (50)")
    plt.xlabel("Episode")
    plt.ylabel("Total reward")
    plt.title("Q-learning on GridWorld")
    plt.legend()
    plt.tight_layout()
    plt.savefig("training_curve.png", dpi=160)
    print("\\nSaved training curve to training_curve.png")

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
import argparse
import numpy as np
import matplotlib.pyplot as plt

class GridWorld:
    """
    5x5 grid with walls (■), a goal (G), a pit (X), and start (S).
    Reward per step = -1, goal = +20 (terminal), pit = -10 (terminal).
    10% 'slip' makes actions stochastic to keep exploration interesting.
    Actions: 0=up, 1=right, 2=down, 3=left.
    """
    def __init__(self, width=5, height=5, start=(0,0), goal=(4,4),
                 walls={(1,1),(1,2),(1,3)}, pits={(3,3)}, slip=0.1, step_cost=-1.0,
                 goal_reward=20.0, pit_reward=-10.0, seed=0):
        self.width = width
        self.height = height
        self.start = start
        self.goal = goal
        self.walls = set(walls)
        self.pits = set(pits)
        self.slip = slip
        self.step_cost = step_cost
        self.goal_reward = goal_reward
        self.pit_reward = pit_reward
        self.rng = np.random.default_rng(seed)
        self.n_actions = 4
        self.n_states = width * height
        self.pos = start

    def state_id(self, pos):
        x, y = pos
        return y * self.width + x

    def id_to_pos(self, sid):
        x = sid % self.width
        y = sid // self.width
        return (x, y)

    def reset(self):
        self.pos = self.start
        return self.state_id(self.pos)

    def apply_action(self, pos, a):
        x, y = pos
        if a == 0:
            y -= 1
        elif a == 1:
            x += 1
        elif a == 2:
            y += 1
        elif a == 3:
            x -= 1
        return (x, y)

    def step(self, a):

            a = self.rng.integers(0, self.n_actions)
        nx, ny = self.apply_action(self.pos, a)
        nx = min(max(nx, 0), self.width-1)
        ny = min(max(ny, 0), self.height-1)
        if (nx, ny) in self.walls:
            nx, ny = self.pos
        self.pos = (nx, ny)

        reward = self.step_cost
        done = False
        if self.pos in self.pits:
            reward = self.pit_reward
            done = True
        elif self.pos == self.goal:
            reward = self.goal_reward
            done = True
        return self.state_id(self.pos), reward, done, {}

    def render_policy(self, Q):
        arrows = {0:'↑',1:'→',2:'↓',3:'←'}
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x,y) in self.walls:
                    row.append('■')
                elif (x,y) in self.pits:
                    row.append('X')
                elif (x,y) == self.goal:
                    row.append('G')
                else:
                    a = int(np.argmax(Q[self.state_id((x,y))]))
                    row.append(arrows[a])
            grid.append(row)
        return grid

def eps_greedy_action(Q_row, eps, rng):
    if rng.random() < eps:
        return rng.integers(0, len(Q_row))
    return int(np.argmax(Q_row))

def sarsa(env, episodes=800, alpha=0.1, gamma=0.99, eps_start=1.0,
          eps_end=0.05, eps_decay=0.995, max_steps=100, seed=0):
    rng = np.random.default_rng(seed)
    Q = np.zeros((env.n_states, env.n_actions), dtype=np.float32)
    eps = eps_start
    rewards = []

    for ep in range(episodes):
        s = env.reset()
        a = eps_greedy_action(Q[s], eps, rng)
        total = 0.0
        for t in range(max_steps):
            s2, r, done, _ = env.step(a)
            total += r
            if done:
                td_target = r
                Q[s, a] += alpha * (td_target - Q[s, a])
                break
            else:
                a2 = eps_greedy_action(Q[s2], eps, rng)
                td_target = r + gamma * Q[s2, a2]
                Q[s, a] += alpha * (td_target - Q[s, a])
                s, a = s2, a2
        rewards.append(total)
        eps = max(eps_end, eps * eps_decay)
    return Q, rewards

def moving_average(x, w):
    x = np.asarray(x, dtype=np.float64)
    if len(x) < w:
        return np.array([])
    return np.convolve(x, np.ones(w)/w, mode='valid')

def rollout(env, Q, max_steps=50):
    s = env.reset()
    path = [env.pos]
    total = 0.0
    for t in range(max_steps):
        a = int(np.argmax(Q[s]))
        s, r, done, _ = env.step(a)
        path.append(env.pos)
        total += r
        if done:
            break
    return path, total, done

def main():
    parser = argparse.ArgumentParser(description="SARSA on a 5x5 GridWorld (tabular)")
    parser.add_argument("--episodes", type=int, default=800)
    parser.add_argument("--alpha", type=float, default=0.1)
    parser.add_argument("--gamma", type=float, default=0.99)
    parser.add_argument("--eps_start", type=float, default=1.0)
    parser.add_argument("--eps_end", type=float, default=0.05)
    parser.add_argument("--eps_decay", type=float, default=0.995)
    parser.add_argument("--slip", type=float, default=0.1)
    parser.add_argument("--seed", type=int, default=0)
    args = parser.parse_args()

    env = GridWorld(slip=args.slip, seed=args.seed)
    Q, rewards = sarsa(env, episodes=args.episodes, alpha=args.alpha, gamma=args.gamma,
                       eps_start=args.eps_start, eps_end=args.eps_end, eps_decay=args.eps_decay)

    print("Learned greedy policy (SARSA). Legend: ■=wall, X=pit, G=goal\\n")
    policy_grid = env.render_policy(Q)
    for row in policy_grid:
        print(" ".join(row))

    path, total, done = rollout(env, Q)
    print("\\nGreedy rollout from S to terminal:")
    print(" -> ".join([str(p) for p in path]))
    print(f"Total reward: {total:.1f} | Reached terminal: {done}")

    ma = moving_average(rewards, 50)
    plt.figure()
    plt.plot(rewards, label="Episode reward")
    if len(ma) > 0:
        start_idx = 50 - 1
        plt.plot(range(start_idx, start_idx + len(ma)), ma, label="Moving average (50)")
    plt.xlabel("Episode")
    plt.ylabel("Total reward")
    plt.title("SARSA on GridWorld")
    plt.legend()
    plt.tight_layout()
    plt.savefig("training_curve_sarsa.png", dpi=160)
    print("\\nSaved training curve to training_curve_sarsa.png")

if __name__ == "__main__":
    main()

Double Q‑learning

Reduces over‑estimation by splitting action selection and evaluation across two tables Q1 and Q2.

def eps_greedy_action(Qsum_row, eps, rng):
    if rng.random() < eps:
        return rng.integers(0, len(Qsum_row))
    return int(np.argmax(Qsum_row))

def double_q_learning(env, episodes=800, alpha=0.1, gamma=0.99, eps_start=1.0,
                      eps_end=0.05, eps_decay=0.995, max_steps=100, seed=0):
    rng = np.random.default_rng(seed)
    Q1 = np.zeros((env.n_states, env.n_actions), dtype=np.float32)
    Q2 = np.zeros((env.n_states, env.n_actions), dtype=np.float32)
    eps = eps_start
    rewards = []

    for ep in range(episodes):
        s = env.reset()
        total = 0.0
        for t in range(max_steps):
            Qsum = Q1 + Q2
            a = eps_greedy_action(Qsum[s], eps, rng)
            s2, r, done, _ = env.step(a)
            total += r

            if done:
                if rng.random() < 0.5:
                    Q1[s, a] += alpha * (r - Q1[s, a])
                else:
                    Q2[s, a] += alpha * (r - Q2[s, a])
            else:
                if rng.random() < 0.5:
                    a_star = int(np.argmax(Q1[s2]))
                    target = r + gamma * Q2[s2, a_star]
                    Q1[s, a] += alpha * (target - Q1[s, a])
                else:
                    a_star = int(np.argmax(Q2[s2]))
                    target = r + gamma * Q1[s2, a_star]
                    Q2[s, a] += alpha * (target - Q2[s, a])

            s = s2
            if done:
                break

        rewards.append(total)
        eps = max(eps_end, eps * eps_decay)

    return Q1, Q2, rewards

def moving_average(x, w):
    x = np.asarray(x, dtype=np.float64)
    if len(x) < w:
        return np.array([])
    return np.convolve(x, np.ones(w)/w, mode='valid')

def rollout(env, Qsum, max_steps=50):
    s = env.reset()
    path = [env.pos]
    total = 0.0
    for t in range(max_steps):
        a = int(np.argmax(Qsum[s]))
        s, r, done, _ = env.step(a)
        path.append(env.pos)
        total += r
        if done:
            break
    return path, total, done

def main():
    parser = argparse.ArgumentParser(description="Double Q-learning on a 5x5 GridWorld (tabular)")
    parser.add_argument("--episodes", type=int, default=800)
    parser.add_argument("--alpha", type=float, default=0.1)
    parser.add_argument("--gamma", type=float, default=0.99)
    parser.add_argument("--eps_start", type=float, default=1.0)
    parser.add_argument("--eps_end", type=float, default=0.05)
    parser.add_argument("--eps_decay", type=float, default=0.995)
    parser.add_argument("--slip", type=float, default=0.1)
    parser.add_argument("--seed", type=int, default=0)
    args = parser.parse_args()

    env = GridWorld(slip=args.slip, seed=args.seed)
    Q1, Q2, rewards = double_q_learning(env, episodes=args.episodes, alpha=args.alpha, gamma=args.gamma,
                                        eps_start=args.eps_start, eps_end=args.eps_end, eps_decay=args.eps_decay,
                                        seed=args.seed)
    Qsum = Q1 + Q2

    print("Learned greedy policy (Double Q-learning). Legend: ■=wall, X=pit, G=goal\\n")
    policy_grid = env.render_policy(Qsum)
    for row in policy_grid:
        print(" ".join(row))

    path, total, done = rollout(env, Qsum)
    print("\\nGreedy rollout from S to terminal:")
    print(" -> ".join([str(p) for p in path]))
    print(f"Total reward: {total:.1f} | Reached terminal: {done}")

    ma = moving_average(rewards, 50)
    plt.figure()
    plt.plot(rewards, label="Episode reward")
    if len(ma) > 0:

# Tiny Q-learning GridWorld (pure NumPy)

A minimal hands-on reinforcement learning lab that runs anywhere — no Gym needed.

## 1) Set up (Conda on Windows)

```powershell
conda create -n rl-lab python=3.11 -y
conda activate rl-lab
pip install numpy matplotlib
```

## 2) Run

Save `qlearning_grid.py` and run:

```powershell
python qlearning_grid.py
```

You’ll see:
- The learned greedy policy as arrows (■=wall, X=pit, G=goal)
- A greedy rollout from start to terminal
- A `training_curve.png` with episode rewards and a moving average

## 3) Play with it

Try edits to build intuition:

- **Deterministic world:** `--slip 0`
- **Short-sighted agent:** lower `--gamma`
- **Faster/slower learning:** tweak `--alpha`
- **More/less exploration:** tweak `--eps_*` flags
- **Harsher penalties:** change step cost / pit reward in `GridWorld`
- **Longer training:** `--episodes 2000`

## 4) Next steps (ideas)

- Convert to **SARSA**: use `Q[s,a] += α [r + γ Q[s', a'] − Q[s,a]]` with the next action `a'` chosen ε-greedily.
- Add more pits/goals or a bigger grid.
- Log episode lengths; plot them.
- Try **Double Q-learning** to reduce overestimation.
- Replace ε-greedy with **softmax** action selection.
